53def parse_model_with_variant(model_str: str) -> Tuple[str, Optional[str]]: 

54 """解析 'model[variant]' 格式，返回 (model, variant)。 

55 

56 例如: 

57 "deepseek-v4-pro[max]" → ("deepseek-v4-pro", "max") 

58 "gemini-3-pro" → ("gemini-3-pro", None) 

59 """ 

60 import re 

61 

62 stripped = model_str.strip() 

63 match = re.match(r"^(.+?)\s*\[(\w+)\]\s*$", stripped) 

64 if match: 

65 return match.group(1).strip(), match.group(2).lower() 

66 return stripped, None

117def _template_to_json(template: Dict[str, set]) -> dict: 

118 """将内部模板格式转为可序列化的 JSON 格式""" 

119 return { 

120 type_label: sorted([list(entry) for entry in entries]) 

121 for type_label, entries in template.items()

125def _template_from_json(data: dict) -> Dict[str, set]: 

126 """将 JSON 格式转为内部模板格式""" 

127 result = {} 

128 for type_label, entries in data.items(): 

129 result[type_label] = {tuple(entry) for entry in entries} 

130 return result

153def check_template_profile(profile: dict, template: Optional[Dict[str, set]] = None) -> bool: 

154 """ 

155 检查 profile 是否满足模板要求： 

156 - 模板中定义的所有条目（agents + categories）都存在 

157 - 每个角色组（强/弱/多模态）内的所有条目使用同一个模型 

158 

159 参数: 

160 profile: 待检查的 profile 

161 template: 要匹配的模板，默认使用当前模板 

162 """ 

163 if not isinstance(profile, dict): 

164 return False 

165 

166 if template is None: 

167 template = load_template() 

168 for type_label, entries in template.items(): 

169 models = set() 

170 for section, key in entries: 

171 entry = profile.get(section, {}).get(key) 

172 if not isinstance(entry, dict) or "model" not in entry: 

173 return False 

174 models.add(entry["model"]) 

175 if len(models) != 1: 

176 return False 

177 

178 return True

181def validate_fallback_config(data: FallbackData) -> Tuple[bool, str]: 

182 """ 

183 验证 fallback 配置的格式是否合法。 

184 

185 校验规则： 

186 - 顶级键必须是模板分类标签（主模型/强模型/中模型/弱模型/多模态模型） 

187 - 每个值必须包含 fallback_models 键 

188 - fallback_models 必须是列表 

189 - 列表中每个对象必须包含 model 键 

190 - 每个链最多 5 个模型 

191 - 链内不允许重复模型 

192 

193 返回: 

194 (True, "") 验证通过 

195 (False, error_message) 验证失败，附带错误描述 

196 """ 

197 valid_categories = set(DEFAULT_TEMPLATE_GROUPS.keys()) 

198 

199 for category, value in data.items(): 

200 if category not in valid_categories: 

201 return False, f"unknown category: {category}" 

202 

203 if not isinstance(value, dict) or "fallback_models" not in value: 

204 return False, f"missing fallback_models in category: {category}" 

205 

206 models = value["fallback_models"] 

207 if not isinstance(models, list): 

208 return False, f"fallback_models must be a list in category: {category}" 

209 

210 if len(models) > 5: 

211 return False, f"too many models in category: {category}, max is 5" 

212 

213 seen: list = [] 

214 for item in models: 

215 if isinstance(item, dict): 

216 if "model" not in item: 

217 return False, f"object item missing 'model' key in category: {category}" 

218 model_str = str(item["model"]) 

219 else: 

220 model_str = str(item) 

221 if model_str in seen: 

222 return False, f"duplicate model in category: {category}" 

223 seen.append(model_str) 

224 

225 return True, ""

228def get_template_summary(profile: dict) -> Tuple[Dict, Dict]: 

229 """ 

230 按模板分组获取摘要。 

231 仅在 profile 通过 check_template_profile 后调用。 

232 返回: (summary, current_models) 

233 - summary: {类型: {模型名: [(section, key), ...]}} 

234 - current_models: {类型: (模型名, variant)} — variant 为 Optional[str] 

235 """ 

236 summary = {} 

237 current_models = {} 

238 

239 template = load_template() 

240 for type_label, entries in template.items(): 

241 section, key = next(iter(entries)) 

242 entry = profile.get(section, {}).get(key, {}) 

243 common_model = entry.get("model", "") 

244 common_variant = entry.get("variant") 

245 

246 summary[type_label] = {common_model: list(entries)} 

247 current_models[type_label] = (common_model, common_variant) 

248 

249 return summary, current_models

252def print_type_summary(summary: Dict, title: str = None, current_models: Dict = None) -> None: 

253 """打印格式化的模型分类摘要。 

254 

255 current_models: {类型: (模型名, variant)} — 用于显示 variant 信息 

256 """ 

257 if title: 

258 print_info(title) 

259 print() 

260 

261 template = load_template() 

262 for t in template: 

263 if t not in summary: 

264 continue 

265 entries = summary[t] 

266 for model, keys in entries.items(): 

267 agent_keys = [k for s, k in keys if s == "agents"] 

268 cat_keys = [k for s, k in keys if s == "categories"] 

269 parts = [] 

270 if agent_keys: 

271 parts.append(f"agents: {', '.join(sorted(agent_keys))}") 

272 if cat_keys: 

273 parts.append(f"categories: {', '.join(sorted(cat_keys))}") 

274 # 显示 model 名称，如果有 variant 则附加 [variant=xxx] 

275 model_display = model 

276 if current_models and t in current_models: 

277 _, variant = current_models[t] 

278 if variant: 

279 model_display = f"{model} [variant={variant}]" 

280 print(f" {Colors.BOLD}{t}{Colors.NC}:") 

281 print(f" {Colors.CYAN}{model_display}{Colors.NC}") 

282 for p in parts: 

283 print(f" {p}") 

284 print()

301def get_fallback_summary(fallback_data: FallbackData) -> Dict[str, List[str]]: 

302 """ 

303 从 fallback 配置中提取每类的显示链。 

304 

305 参数: 

306 fallback_data: fallback 配置字典，格式如: 

307 {"主模型": {"fallback_models": ["model-a", ...]}, ...} 

308 

309 返回: 

310 {category_label: [model_strings]} 

311 每个 model_string 已格式化（字典条目显示为 model [variant=xxx]） 

312 """ 

313 summary: Dict[str, List[str]] = {} 

314 for category, value in fallback_data.items(): 

315 models = value.get("fallback_models", []) if isinstance(value, dict) else [] 

316 summary[category] = [_format_fallback_item(item) for item in models] 

317 return summary

347def sync_profiles_after_template_change( 

348 old_template: Dict[str, set], new_template: Dict[str, set] 

349) -> int: 

350 """同步所有 profiles 到新模板。返回更新的 profile 数量。""" 

351 config = load_config() 

352 updated = 0 

353 

354 old_entry_to_group = {} 

355 for group, entries in old_template.items(): 

356 for entry in entries: 

357 old_entry_to_group[entry] = group 

358 

359 for name in config.get("profiles", {}): 

360 profile = load_profile_json(name) 

361 if profile is None: 

362 continue 

363 

364 # 跳过不满足旧模板的 profile（如 --detail 模式创建的）， 

365 # 避免对这些 profile 误加空条目并错误计数 

366 if not check_template_profile(profile, old_template): 

367 continue 

368 

369 changed = False 

370 for group, entries in new_template.items(): 

371 for section, key in entries: 

372 if section not in profile: 

373 profile.setdefault(section, {}) 

374 if key not in profile[section]: 

375 profile[section][key] = {"model": ""} 

376 changed = True 

377 

378 old_group = old_entry_to_group.get((section, key)) 

379 if old_group and old_group != group: 

380 old_model = profile.get(section, {}).get(key, {}).get("model", "") 

381 group_models = set() 

382 for s, k in new_template.get(group, set()): 

383 m = profile.get(s, {}).get(k, {}).get("model") 

384 if m: 

385 group_models.add(m) 

386 group_models.discard("") 

387 if group_models: 

388 profile[section][key]["model"] = sorted(group_models)[0] 

389 else: 

390 profile[section][key]["model"] = old_model 

391 changed = True 

392 

393 if changed: 

394 profile_path = get_profile_path(name) 

395 _create_version_snapshot(profile_path, "sync_profiles") 

396 _atomic_write_json(profile_path, profile) 

397 _rotate_versions(profile_path) 

398 updated += 1 

399 

400 return updated

403def cmd_template(args: List[str]) -> None: 

404 """查看或编辑模板""" 

405 # 延迟导入避免循环依赖 

406 from .cli import open_editor 

407 

408 if not args: 

409 template = load_template() 

410 is_custom = TEMPLATE_FILE.exists() 

411 

412 if is_custom: 

413 print_info("当前模板 (自定义):") 

414 else: 

415 print_info("当前模板 (默认):") 

416 

417 print() 

418 for type_label, entries in template.items(): 

419 agent_keys = sorted([k for s, k in entries if s == "agents"]) 

420 cat_keys = sorted([k for s, k in entries if s == "categories"]) 

421 print(f" {Colors.BOLD}{type_label}{Colors.NC}:") 

422 if agent_keys: 

423 print(f" agents: {', '.join(agent_keys)}") 

424 if cat_keys: 

425 print(f" categories: {', '.join(cat_keys)}") 

426 print() 

427 

428 print_dim(f"模板文件: {TEMPLATE_FILE}") 

429 if is_custom: 

430 print_dim("使用 'oma-switch template reset' 恢复默认模板") 

431 else: 

432 print_dim("使用 'oma-switch template edit' 自定义模板") 

433 return 

434 

435 subcommand = args[0] 

436 

437 if subcommand == "edit": 

438 template = load_template() 

439 json_data = _template_to_json(template) 

440 

441 CONFIG_DIR.mkdir(parents=True, exist_ok=True) 

442 _create_version_snapshot(TEMPLATE_FILE, "cmd_template") 

443 _atomic_write_json(TEMPLATE_FILE, json_data) 

444 _rotate_versions(TEMPLATE_FILE) 

445 

446 print_info("正在编辑模板...") 

447 if open_editor(TEMPLATE_FILE): 

448 if is_valid_json(TEMPLATE_FILE): 

449 try: 

450 with open(TEMPLATE_FILE, "r", encoding="utf-8") as f: 

451 new_data = json.load(f) 

452 new_template = _template_from_json(new_data) 

453 

454 updated = sync_profiles_after_template_change(template, new_template) 

455 save_template(new_template) 

456 

457 print_success(f"模板已更新，同步了 {updated} 个配置文件") 

458 except (json.JSONDecodeError, KeyError, ValueError, IOError) as e: 

459 print_error(f"模板格式错误: {e}") 

460 if template != DEFAULT_TEMPLATE_GROUPS: 

461 save_template(template) 

462 print_info("已恢复之前的模板") 

463 else: 

464 TEMPLATE_FILE.unlink(missing_ok=True) 

465 print_info("已恢复默认模板") 

466 else: 

467 print_error("模板文件格式错误，已恢复") 

468 if template != DEFAULT_TEMPLATE_GROUPS: 

469 save_template(template) 

470 else: 

471 TEMPLATE_FILE.unlink(missing_ok=True) 

472 else: 

473 print_error("编辑失败") 

474 return 

475 

476 if subcommand == "reset": 

477 if TEMPLATE_FILE.exists(): 

478 old_template = load_template() 

479 try: 

480 updated = sync_profiles_after_template_change(old_template, DEFAULT_TEMPLATE_GROUPS) 

481 TEMPLATE_FILE.unlink() 

482 print_success(f"已恢复默认模板，同步了 {updated} 个配置文件") 

483 except IOError as e: 

484 print_error(f"同步配置文件失败: {e}") 

485 print_warning("模板文件未删除，配置可能不一致") 

486 else: 

487 print_info("当前使用的就是默认模板") 

488 return 

489 

490 if subcommand == "diff": 

491 template = load_template() 

492 is_custom = TEMPLATE_FILE.exists() 

493 

494 if not is_custom: 

495 print_info("当前使用默认模板，没有差异可比较") 

496 return 

497 

498 print_info("自定义模板 vs 默认模板:") 

499 print() 

500 

501 all_groups = set(template.keys()) | set(DEFAULT_TEMPLATE_GROUPS.keys()) 

502 for group in sorted(all_groups): 

503 custom_entries = template.get(group, set()) 

504 default_entries = DEFAULT_TEMPLATE_GROUPS.get(group, set()) 

505 

506 added = custom_entries - default_entries 

507 removed = default_entries - custom_entries 

508 

509 if not added and not removed: 

510 continue 

511 

512 print(f" {Colors.BOLD}{group}{Colors.NC}:") 

513 for section, key in sorted(added): 

514 print(f" {Colors.GREEN}+ {section}.{key}{Colors.NC}") 

515 for section, key in sorted(removed): 

516 print(f" {Colors.RED}- {section}.{key}{Colors.NC}") 

517 print() 

518 return 

519 

520 print_error(f"未知子命令: {subcommand}") 

521 print_info("用法: oma-switch template [edit|reset|diff]") 

522 sys.exit(1)
