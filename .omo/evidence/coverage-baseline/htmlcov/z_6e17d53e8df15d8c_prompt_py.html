25 current 可以是 (model, variant) 元组或纯字符串（向后兼容）。 

26 用户可以通过编号选择现有模型，也可以输入 model[variant] 格式。 

27 支持模糊搜索：输入非数字文本时过滤模型列表。 

28 

29 all_models 参数保留向后兼容：如果提供且非空则使用，否则使用 collect_models_enriched。 

30 """ 

31 if isinstance(current, tuple): 

32 current_model, current_variant = current 

33 else: 

34 current_model, current_variant = current, None 

35 

36 if all_models: 

37 enriched_models: List[Tuple[str, Optional[str], int]] = [ 

38 (m, None, get_model_frequency(m)) for m in all_models 

39 ] 

40 else: 

41 enriched_models = collect_models_enriched(category=type_label)

43 if not enriched_models:

116def prompt_select_fallback_models( 

117 type_label: str, 

118 all_models: List[str], 

119 current: Optional[List] = None, 

120) -> List: 

121 """交互式提示用户选择 fallback 模型链，返回选中的模型列表。 

122 

123 current 可以是字符串列表或包含 variant 的字典列表（向后兼容）。 

124 用户可以通过逗号分隔的编号或模型名选择多个模型，支持 model[variant] 格式。 

125 空输入在有当前值时保留当前链，在无当前值时清空链（返回 []）。 

126 最多选择 MAX_FALLBACK_MODELS 个模型。 

127 模型按分类分组展示，支持搜索过滤，记录使用历史。 

128 """ 

129 if current is None: 

130 current = [] 

131 

132 print(f" {Colors.BOLD}{type_label} Fallback 链{Colors.NC}") 

133 if current: 

134 display_items = [] 

135 for item in current: 

136 if isinstance(item, dict): 

137 m = item.get("model", "") 

138 v = item.get("variant") 

139 display_items.append(f"{m}[{v}]" if v else m) 

140 else: 

141 display_items.append(str(item)) 

142 print(f" 当前: {Colors.CYAN}{', '.join(display_items)}{Colors.NC}") 

143 else: 

144 print(f" 当前: {Colors.GRAY}(空){Colors.NC}") 

145 

146 enriched = collect_models_enriched(type_label) 

147 if not enriched: 

148 enriched = [(m, None, 0) for m in all_models] 

149 model_names = [m for m, _v, _f in enriched] 

150 

151 last_used = None 

152 best_freq = 0 

153 for model, _v, _f in enriched: 

154 cat_freq = get_category_frequency(model, type_label) 

155 if cat_freq > best_freq: 

156 best_freq = cat_freq 

157 last_used = model

159 def _display_grouped(models_enriched: List[Tuple[str, Optional[str], int]]) -> List[str]: 

160 print(f" 可用模型（最多选 {MAX_FALLBACK_MODELS} 个）:") 

161 indexed: List[str] = [] 

162 idx = 1 

163 categorized_set: set = set() 

164 

165 for category in DEFAULT_TEMPLATE_GROUPS: 

166 cat_models = [ 

167 (m, v, f) for m, v, f in models_enriched if get_category_frequency(m, category) > 0 

168 ] 

169 if cat_models: 

170 print(f" {Colors.GRAY}── {category} ──{Colors.NC}") 

171 for model, variant, _freq in cat_models: 

172 var_str = f" [{variant}]" if variant else ""

174 print(f" [{idx}] {model}{var_str}{last_str}") 

175 indexed.append(model) 

176 categorized_set.add(model)

178

180 if other_models: 

181 print(f" {Colors.GRAY}── 其他 ──{Colors.NC}") 

182 for model, variant, _freq in other_models: 

183 var_str = f" [{variant}]" if variant else "" 

184 last_str = f" {Colors.YELLOW}(上次){Colors.NC}" if model == last_used else "" 

185 print(f" [{idx}] {model}{var_str}{last_str}") 

186 indexed.append(model) 

187 idx += 1 

188 

189 return indexed

191 def _apply_selection(parts: List[str], indexed: List[str]) -> List: 

192 selected: List = [] 

193 for part in parts: 

194 if part.isdigit(): 

195 i = int(part) - 1 

196 if 0 <= i < len(indexed): 

197 selected.append(indexed[i]) 

198 else: 

199 print_warning(f"无效编号: {part}，已跳过") 

200 else: 

201 model, variant = parse_model_with_variant(part) 

202 if variant: 

203 selected.append({"model": model, "variant": variant}) 

204 else: 

205 selected.append(model) 

206 return selected

208 def _truncate_and_record(selected: List) -> List: 

209 if len(selected) > MAX_FALLBACK_MODELS: 

210 print_warning( 

211 f"最多只能选择 {MAX_FALLBACK_MODELS} 个 fallback 模型，" 

212 f"已截取前 {MAX_FALLBACK_MODELS} 个" 

213 ) 

214 selected = selected[:MAX_FALLBACK_MODELS] 

215 for item in selected: 

216 m = item.get("model", item) if isinstance(item, dict) else item 

217 record_model_usage(m, "fallback") 

218 return selected

220 indexed_models = _display_grouped(enriched)

222 if current: 

223 prompt_text = ( 

224 f" 请输入{type_label} fallback 模型（逗号分隔的编号/模型名[variant]，留空=保留当前）: " 

225 ) 

226 else: 

227 prompt_text = ( 

228 f" 请输入{type_label} fallback 模型（逗号分隔的编号/模型名[variant]，留空=清空）: " 

229 ) 

230 

231 while True: 

232 choice = input(prompt_text).strip() 

233 

234 if not choice: 

235 return list(current) if current else [] 

236 

237 parts = [p.strip() for p in choice.split(",") if p.strip()] 

238 

239 if len(parts) == 1 and not parts[0].isdigit(): 

240 query = parts[0] 

241 if any(c in query for c in "-.[]/"): 

242 model, variant = parse_model_with_variant(query) 

243 selected = [{"model": model, "variant": variant}] if variant else [model] 

244 return _truncate_and_record(selected) 

245 

246 matches = fuzzy_match_models(query, model_names) 

247 if matches: 

248 print(f" {Colors.GRAY}搜索 '{query}' 的结果:{Colors.NC}") 

249 match_set = set(matches) 

250 filtered = [(m, v, f) for m, v, f in enriched if m in match_set] 

251 filtered_indexed = _display_grouped(filtered) 

252 

253 inner = input(" 请选择（逗号分隔编号，留空=取消搜索）: ").strip()

259 selected = _apply_selection(inner_parts, filtered_indexed) 

260 return _truncate_and_record(selected) 

261 else: 

262 print_warning(f"未找到匹配 '{query}' 的模型") 

263 continue 

264 

265 selected = _apply_selection(parts, indexed_models) 

266 return _truncate_and_record(selected)

269def generate_profile_from_types( 

270 template: dict, 

271 model_map: Dict[str, Tuple[str, Optional[str]]], 

272) -> dict: 

273 """ 

274 根据模板生成新 profile。 

275 model_map: {类型: (新模型名, variant)} 

276 按模板分组替换各角色的 model 和 variant。 

277 如果模板中的条目在 profile 中不存在，则自动创建。 

278 """ 

279 new_profile = copy.deepcopy(template) 

280 tpl = load_template() 

281 for type_label, entries in tpl.items(): 

282 model_info = model_map.get(type_label) 

283 if model_info: 

284 new_model, variant = model_info 

285 for section, key in entries: 

286 if section not in new_profile: 

287 new_profile[section] = {} 

288 if key not in new_profile[section]: 

289 new_profile[section][key] = {} 

290 

291 new_profile[section][key]["model"] = new_model 

292 if variant: 

293 new_profile[section][key]["variant"] = variant 

294 else: 

295 new_profile[section][key].pop("variant", None) 

296 return new_profile

299def generate_fallback_from_types( 

300 fallback_choices: Dict[str, List], 

301) -> FallbackData: 

302 """ 

303 根据用户选择的 fallback 模型列表生成完整的 fallback 配置字典。 

304 fallback_choices: {类型标签: [模型名或{model, variant}字典, ...]} 

305 按模板结构输出: {类型标签: {"fallback_models": [...]}} 

306 未指定的类型默认为 {"fallback_models": []}。 

307 """ 

308 tpl = load_template() 

309 result: FallbackData = {} 

310 for type_label in tpl: 

311 chain = fallback_choices.get(type_label, []) 

312 result[type_label] = {"fallback_models": list(chain)} # type: ignore[literal-required] 

313 return result
